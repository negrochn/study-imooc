```js
// 主题，保存状态，状态变化之后触发所有观察者对象
class Subject {
  constructor() {
    this.state = 0
    this.observers = []
  }
  getState() {
    return this.state
  }
  setState(state) {
    this.state = state
    this.notifyAllObservers()
  }
  notifyAllObservers() {
    this.observers.forEach(observer => {
      observer.update()
    })
  }
  attach(observer) {
    this.observers.push(observer)
  }
}

// 观察者
class Observer {
  constructor(name, subject) {
    this.name = name
    this.subject = subject
    this.subject.attach(this)
  }
  update() {
    console.log(`${this.name} update, state: ${this.subject.getState()}`)
  }
}

// 测试
let s = new Subject()
let o1 = new Observer('o1', s)
let o2 = new Observer('o2', s)
let o3 = new Observer('o3', s)

s.setState(1)
s.setState(2)
```

Promise

```js
function loadImg(src) {
  return new Promise(function(resolve, reject) {
    const img = document.createElement('img')
    img.onload = function() {
      resolve(img)
    }
    img.onerror = function() {
      reject('图片加载失败')
    }
    img.src = src
  })
}
```


nodejs 自定义事件

```js
const { EventEmitter } = require('events')

const emitter1 = new EventEmitter()
emitter1.on('some', () => {
  // 监听 some 事件
  console.log('some event is occured 1')
})
emitter1.on('some', () => {
  // 监听 some 事件
  console.log('some event is occured 2')
})
// 触发 some 事件
emitter1.emit('some')

emitter1.on('showName', name => {
  console.log(name, 'event occured')
})
emitter1.emit('showName', 'Tik Tok')
```

```js
const { EventEmitter } = require('events')

// 任何构造函数都可以继承 EventEmitter 的方法 on emit
class Dog extends EventEmitter {
  constructor(name) {
    super()
    this.name = name
  }
}

const wangcai = new Dog('wangcai')
wangcai.on('bark', function() {
  console.log(this.name, 'barked')
})
setInterval(() => {
  wangcai.emit('bark')
}, 2000)
```

```js
// 注意：需要进入 es6/src ，执行 node index.js

const fs = require('fs')
const readStream = fs.createReadStream('./index.js')

let length = 0
readStream.on('data', chunk => {
  length += chunk.toString().length
})
readStream.on('end', () => {
  console.log(length)
})
```

```js
const fs = require('fs')
const readline = require('readline')

const rl = readline.createInterface({
  input: fs.createReadStream('./index.html')
})

let lineNum = 0
rl.on('line', (line) => {
  console.log(line)
  lineNum++
})
rl.on('close', () => {
  console.log(lineNum)
})
```
